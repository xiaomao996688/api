108 #if defined(CPPGC_CAGED_HEAP)
109 class WriteBarrierTypeForCagedHeapPolicy final {
110  public:
111  template <WriteBarrier::ValueMode value_mode>
112  static V8_INLINE WriteBarrier::Type Get(const void* slot, const void* value,
113  WriteBarrier::Params& params) {
114  const bool have_caged_heap =
115  value_mode == WriteBarrier::ValueMode::kValuePresent
116  ? TryGetCagedHeap(slot, value, params)
117  : TryGetCagedHeap(slot, slot, params);
118  if (!have_caged_heap) {
119  return WriteBarrier::Type::kNone;
120  }
121  if (V8_UNLIKELY(params.caged_heap().is_marking_in_progress)) {
122 #if V8_ENABLE_CHECKS
123  params.type = WriteBarrier::Type::kMarking;
124 #endif // !V8_ENABLE_CHECKS
125  return WriteBarrier::Type::kMarking;
126  }
127 #if defined(CPPGC_YOUNG_GENERATION)
128  params.slot_offset = reinterpret_cast<uintptr_t>(slot) - params.start;
129  if (value_mode == WriteBarrier::ValueMode::kValuePresent) {
130  params.value_offset = reinterpret_cast<uintptr_t>(value) - params.start;
131  } else {
132  params.value_offset = 0;
133  }
134 #if V8_ENABLE_CHECKS
135  params.type = WriteBarrier::Type::kGenerational;
136 #endif // !V8_ENABLE_CHECKS
137  return WriteBarrier::Type::kGenerational;
138 #else // !CPPGC_YOUNG_GENERATION
139  return WriteBarrier::Type::kNone;
140 #endif // !CPPGC_YOUNG_GENERATION
141  }
142 
143  private:
144  WriteBarrierTypeForCagedHeapPolicy() = delete;
145 
146  static V8_INLINE bool TryGetCagedHeap(const void* slot, const void* value,
147  WriteBarrier::Params& params) {
148  params.start = reinterpret_cast<uintptr_t>(value) &
149  ~(api_constants::kCagedHeapReservationAlignment - 1);
150  const uintptr_t slot_offset =
151  reinterpret_cast<uintptr_t>(slot) - params.start;
152  if (slot_offset > api_constants::kCagedHeapReservationSize) {
153  // Check if slot is on stack or value is sentinel or nullptr. This relies
154  // on the fact that kSentinelPointer is encoded as 0x1.
155  return false;
156  }
157  return true;
158  }
159 };
160 #endif // CPPGC_CAGED_HEAP
