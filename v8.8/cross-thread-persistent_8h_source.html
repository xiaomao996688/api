220  void Assign(T* ptr) {
221  void* old_value = GetValue();
222  if (IsValid(old_value)) {
223  PersistentRegionLock guard;
224  PersistentRegion& region = this->GetPersistentRegion(old_value);
225  if (IsValid(ptr) && (&region == &this->GetPersistentRegion(ptr))) {
226  SetValue(ptr);
227  this->CheckPointer(ptr);
228  return;
229  }
230  region.FreeNode(GetNode());
231  SetNode(nullptr);
232  }
233  SetValue(ptr);
234  if (!IsValid(ptr)) return;
235  PersistentRegionLock guard;
236  SetNode(this->GetPersistentRegion(ptr).AllocateNode(this, &Trace));
237  this->CheckPointer(ptr);
238  }

244  if (IsValid(ptr) && (&region == &this->GetPersistentRegion(ptr))) {

253  if (!IsValid(ptr)) return;
